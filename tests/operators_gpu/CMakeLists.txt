if(NOT USE_GPU)
    return()

set(operator_test operators_gpu)

foreach(item ${operator_test})
    add_executable(${item} ${item}.cu ${SOURCES} ${GPU_SOURCES})
    add_dependencies(${item} copy_files)
    target_link_libraries(${item} iblgf_test_main
        ${CUDA_CUFFT_LIBRARIES} cufft)
    set_target_properties(${item} PROPERTIES
        CUDA_SEPARABLE_COMPILATION ON
        CUDA_STANDARD 17
    )
    add_test(
        NAME ${item}
        COMMAND ${MPIEXEC_EXECUTABLE} ${MPIEXEC_PREFLAGS} ${MPIEXEC_NUMPROC_FLAG} 2 $<TARGET_FILE:${item}>
    )
endforeach()
